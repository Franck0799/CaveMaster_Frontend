  private initializeAvailabilityOptions(): void {
    setTimeout(() => {
      const options = document.querySelectorAll('.availability-option');

        option.addEventListener('click', () => {

          const text = option.getAttribute('data-text') || '';
          this.changeAvailability(status, icon, text);
        });
      });
    }, 0);
  }

  addToOrder(itemId: number): void {

  calculateOrderTotal(): void {
    this.orderTotal = this.orderItems.reduce((sum, item) => sum + item.price, 0);
  }

  createOrder(): void {
    if (!this.selectedTable) {
      this.showNotification('‚ö†Ô∏è Veuillez s√©lectionner une table');
      return;
    }

    if (this.orderItems.length === 0) {
      this.showNotification('‚ö†Ô∏è Veuillez ajouter au moins un produit');
      return;
    }

    const newOrder: Order = {
      table: parseInt(this.selectedTable),
      status: 'new',
      statusText: 'Nouvelle',
      time: '0 min',
      items: this.orderItems.map(item => ({
        name: item.name,
        price: item.price
      })),
      total: this.orderTotal
    };

    this.orders.push(newOrder);
    this.selectedTable = '';
    this.guestCount = 2;
    this.orderItems = [];
    this.orderTotal = 0;
    this.showNotification(`‚úì Nouvelle commande cr√©√©e pour la Table #${newOrder.table}`);
    this.navigateTo('orders');
    console.log('Nouvelle commande cr√©√©e:', newOrder);
  }

  updateTimeElapsed(): void {
    const startTime = new Date();
    startTime.setHours(8, 0, 0);

    setInterval(() => {
      const now = new Date();
      const diff = now.getTime() - startTime.getTime();
      const hours = Math.floor(diff / (1000 * 60 * 60));
      const minutes = Math.floor((diff % (1000 * 60 * 60)) / (1000 * 60));
      const timeStat = this.stats.find(s => s.label === 'Temps √©coul√©');
      if (timeStat) {
        timeStat.value = `${hours}h ${minutes}m`;
      }
    }, 60000);
  }
